const preview=document.getElementById("preview"); const thumbnails=document.querySelectorAll("#thumbnails img"); const prevBtn=document.getElementById("prev"); const nextBtn=document.getElementById("next"); let current=0; function showImage(index){ current=index; preview.src=thumbnails[index].src; thumbnails.forEach(t=>t.classList.remove("active")); thumbnails[index].classList.add("active"); } prevBtn.onclick=()=>{ current=(current-1+thumbnails.length)%thumbnails.length; showImage(current); }; nextBtn.onclick=()=>{ current=(current+1)%thumbnails.length; showImage(current); }; thumbnails.forEach((thumb,i)=>{ thumb.onclick=()=>showImage(i); });
